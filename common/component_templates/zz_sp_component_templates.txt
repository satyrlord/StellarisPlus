# zz_sp_ prefix: LIOS override for vanilla combat-computer auto-design bug (4.0+).
# COMBAT_COMPUTER_DEFAULT has upgrade_path = default with no upgrades_to chain, so
# auto-designed ships never upgrade combat computers. Hide it and restore role defaults.

@cost0 = 0
@power0 = -5
@computer_cost0 = 0
@computer_power0 = -5
@alloy_to_food_cost_ratio = 3.5

utility_component_template = {
	key = "COMBAT_COMPUTER_DEFAULT"
	size = small
	icon = "GFX_ship_part_computer_default"
	icon_frame = 1
	ship_behavior = "default"
	upgrade_path = default
	hidden = yes
	potential = {
		from = {
			country_uses_bio_ships = no
		}
		OR = {
			is_ship_class = shipclass_military
			is_ship_class = shipclass_entropy_conduit
			is_ship_size = juggernaut
		}
	}

	show_tech_unlock_if = {
		country_uses_bio_ships = no
	}
	component_set = "combat_computers"

	ai_weight = {
		weight = 1
	}
}
