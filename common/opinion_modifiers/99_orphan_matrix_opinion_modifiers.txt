triggered_opinion_guardian_matrix_bad_crisis_to_custodian_matrix = {
	trigger = {
		FROM = {
			is_country_type = default
			exists = ruler
			ruler = { has_trait = leader_trait_orphan_matrix_galactic_guardianship }
		}
		has_crisis_level = crisis_level_2
		OR = {
			has_crisis_level = crisis_level_5
			FROM = {
				has_intel_level = {
					who = ROOT
					category = government
					level >= 2
				}
			}
		}
	}

	opinion = {
		base = -60

		modifier = {
			add = -80
			has_crisis_level = crisis_level_3
		}
		modifier = {
			add = -100
			has_crisis_level = crisis_level_4
		}
		modifier = {
			add = -160 #-400 total
			has_crisis_level = crisis_level_5
		}
	}
}

triggered_opinion_guardian_matrix_good = {
	trigger = {
		FROM = {
			exists = ruler
			ruler = { has_trait = leader_trait_orphan_matrix_galactic_guardianship }
		}
		is_galactic_community_member = yes
		NOT = { has_crisis_level = crisis_level_2 }
		NOT = { is_at_war_with = FROM }
		NOT = { has_truce = FROM }
	}

	opinion = {
		base = 100
	}
}
