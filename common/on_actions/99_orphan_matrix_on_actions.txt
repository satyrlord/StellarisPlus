# Triggers when the game starts
on_game_start = {
	events = {
		# Set-up
		orphan_matrix_events.0
	}
}

# No scope, like on_game_start
on_monthly_pulse = {
	events = {
		# Manage Energy Deficits
		orphan_matrix_events.500
		
		# Attempt to find the Precursor Home system !
		orphan_matrix_events.50
		
		# Attempt to allow Custodian Matrix and Assimilator Matrix to revive the Designers
		orphan_matrix_events.100
		
		# Attempt to rebuilt the Great Sanctuary when the Custodian Matrix restores the Relic World of its Designers
		# If the Vats event hasn't happened yet, carry the Vats on the Ecumenopolis
		orphan_matrix_events.103
		
		# Attempt to give the Custodian Matrix a special anti-crisis trait
		orphan_matrix_events.105
		
		# Attempt to repair the shield on the relic world after the assimilator matrix have had its Designers rebirthed
		orphan_matrix_events.107
		
		# Offer the Memorialist Orphan Matrix the Memorial Project
		orphan_matrix_events.110
		
		# Carry out Terraforming Capital with Lab on it
		orphan_matrix_events.120
		
		# Spot Assimilator assimilating pops and give +% credits - Spot
		# orphan_matrix_events.130 - DISCONTINUED AFTER POP CHANGE (4.0)
		
		# Spot Assimilator assimilating pops and give +% credits - Handling
		orphan_matrix_events.131
		
		# Spot amount of Chroniclers within each Memorialist Empire
		orphan_matrix_events.142
	}
}

# Triggers event when each country first sends a ship into the system (once per country)
# Scope = Ship
# From = System
# FromFrom = Country
on_entering_system_first_time = {
	events = {
		# The Orphaned Matrix enters their Precursor home system for the first time
		orphan_matrix_events.52
	}
}

# A ship has surveyed a planet.
# Scope = Ship
# From = Planet
on_survey_planet = {
	events = {
		# The Orphaned Matrix scans their Designers home planet for the first time
		orphan_matrix_events.60
		
		# The Orphaned Matrix scans the Ruined Arc Furnace in their Designers home system
		orphan_matrix_events.53
	}
}

# An empire is destroyed
#This = destroyed country
#From = optional, destroyer (country)
on_country_destroyed = {
	events = {
		# An empire gets destroyed, as such, the Orphaned Matrix having taken the Memorialist path will get their reward
		orphan_matrix_events.140
	}
}

# A Megastructure has been upgraded
# Root = Country
# From = Megastructure
# FromFrom = System
on_megastructure_upgraded = {
	events = {
		# The Orphaned Matrix has restored its Arc Furnace
		orphan_matrix_events.54
	}
}
