				set_star_flag = eotf_ancient_legacy_chain_system
			}

		limit = {
			planet = { has_planet_flag = eotf_ancient_legacy_burial_world }
		}
		solar_system = {
			set_star_flag = eotf_ancient_legacy_chain_system
		}
	}
}

# One debris cluster on a random planet in the home system (requires event_target:eotf_orbital_debris_homeworld_capital).
# Divine Hulk origin only — called from eotf_spawn_divine_hulk_homeworld_debris.
eotf_spawn_one_divine_hulk_orbital_debris_field = {
	if = {
		limit = {
			any_system_planet = {
				is_star = no
				NOT = { is_same_value = event_target:eotf_orbital_debris_homeworld_capital }
				NOT = { has_planet_flag = eotf_divine_hulk_debris_host }
			}
		}
		random_system_planet = {
			limit = {
				is_star = no
				NOT = { is_same_value = event_target:eotf_orbital_debris_homeworld_capital }
				NOT = { has_planet_flag = eotf_divine_hulk_debris_host }
			}
			set_planet_flag = eotf_divine_hulk_debris_host
			eotf_spawn_orbital_debris_visuals_on_this = yes
		}
	}
	else_if = {
		limit = {
			any_system_planet = {
				is_star = no
				NOT = { has_planet_flag = eotf_divine_hulk_debris_host }
			}
		}
		random_system_planet = {
			limit = {
				is_star = no
				NOT = { has_planet_flag = eotf_divine_hulk_debris_host }
			}
			set_planet_flag = eotf_divine_hulk_debris_host
			eotf_spawn_orbital_debris_visuals_on_this = yes
		}
	}
	else = {
		random_system_planet = {
			limit = {
				is_star = no
			}
			eotf_spawn_orbital_debris_visuals_on_this = yes
		}
	}
}

# Scion Origin Expanded — scripted effects (namespace: soe)

soe_apply_parental_supervision_opinion_to_country = {
	if = {
		limit = {
			exists = $TARGET$
			root = { has_modifier = soe_parental_supervision }
			$TARGET$ = {
				is_country_type = default
				NOT = { is_same_value = root }
				NOT = {
					has_opinion_modifier = {
						who = root
						modifier = opinion_soe_parental_supervision
					}
				}
			}
		}
		$TARGET$ = {
			add_opinion_modifier = {
				who = root
				modifier = opinion_soe_parental_supervision
			}
		}
	}
}

				who = root
				modifier = opinion_soe_parental_supervision
			}
		}
	}
}

# Keep the shared helper aligned with vanilla so non-random FE building grants,
# future upstream uses, and other mods do not inherit the Ancient Legacy gate.
grant_fe_building = {
	change_variable = { which = $BUILDING$_build_count value = 1 }
	change_variable = { which = $BUILDING$_upgrade_count value = 1 }
	set_country_flag = $BUILDING$_can_build
	set_timed_country_flag = { flag = recently_got_$BUILDING$ days = 30 }
	create_message = {
		type = FALLEN_EMPIRE_BUILDING_GAINED
		localization = MESSAGE_FALLEN_EMPIRE_BUILDING_GAINED
		target = root
		days = 30
	}
}
