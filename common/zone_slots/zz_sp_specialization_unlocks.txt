# OVERRIDE: zz_sp_ — slot_food overridden to allow machine empire zone access.
# Vanilla slot_food requires food_processing_1 tech, anglers, or agrarian idyll.
# Added machine empire bypass so retrofit zones (energy/minerals) are accessible.
slot_food = {
	included_zone_sets = {
		food
	}
	unlock = {
		custom_tooltip = {
			fail_text = zone_unavailable_tech_food_processing_1
			OR = {
				AND = {
					exists = owner
					owner = { is_machine_empire = yes }
				}
				AND = {
					exists = owner
					owner = { is_anglers_empire = yes }
					is_wet = yes
				}
				AND = {
					exists = owner
					owner = { has_valid_civic = civic_agrarian_idyll }
				}
				AND = {
					exists = owner
					owner = {
						has_tech_or_functional_civic = {
							TECH = tech_food_processing_1
						}
					}
				}
			}
		}
	}
}
