# Colony type planet modifier inline script
# $MODIFIER$ is the planet modifier to apply (e.g. planet_jobs_produces_mult)
planet_modifier = {
	$MODIFIER$ = 0.10
}

triggered_planet_modifier = {
	potential = {
		exists = owner_or_space_owner
		owner_or_space_owner = { has_tradition = tr_adaptability_finish }
	}
	modifier = { $MODIFIER$ = 0.05 }
}
