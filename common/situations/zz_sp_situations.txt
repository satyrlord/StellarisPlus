# Divine Hulk — Questioning the Divine (empire-wide; begins from eotf.255 option).

situation_eotf_questioning_the_divine = {
	picture = GFX_eotf_evt_divine_hulk
	category = neutral

	complete_icon_frame = GFX_situation_outcome_frame_blue
	complete_icon = GFX_situation_outcome_positive
	fail_icon_frame = GFX_situation_outcome_frame_red
	fail_icon = GFX_situation_outcome_negative

	progress_direction = monodirectional

	abort_trigger = {
		OR = {
			NOT = { exists = owner }
			owner = {
				OR = {
					NOT = { has_origin = origin_divine_hulk }
					is_gestalt = yes
				}
			}
		}
	}

	on_abort = {
		hidden_effect = {
			destroy_situation = this
		}
	}

	# Ensure a valid approach is active (default alone is not always applied on start_situation).
	on_start = {
		set_situation_approach = approach_eotf_dh_measured_inquiry
	}

	on_progress_complete = {
		custom_tooltip = situation_eotf_questioning_the_divine_complete_tt
		hidden_effect = {
			owner = {
				if = {
					limit = {
						is_ai = no
					}
					country_event = {
						id = eotf.262
					}
				}
				else = {
					if = {
						limit = {
							capital_scope = {
								NOT = { exists = archaeological_site }
								NOT = { has_planet_flag = suppress_archaeological_sites }
							}
						}
						capital_scope = {
							create_archaeological_site = eotf_site_divine_hulk
						}
					}
				}
			}
			destroy_situation = this
		}
	}

	# monthly_progress: base always runs; approach mults adjust speed. Total progress 0–300 (final stage end).
	monthly_progress = {
		base = 0.6
		modifier = {
			mult = 0.85
			desc = situation_eotf_questioning_the_divine_progress_measured
			current_situation_approach = approach_eotf_dh_measured_inquiry
		}
		modifier = {
			mult = 1.2
			desc = situation_eotf_questioning_the_divine_progress_forceful
			current_situation_approach = approach_eotf_dh_forceful_inquiry
		}
	}

	stages = {
		stage_eotf_dh_1 = {
			icon = GFX_situation_stage_1
			icon_background = GFX_situation_stage_frame_green
			end = 100
			custom_tooltip = stage_eotf_dh_1_tt
		}
		stage_eotf_dh_2 = {
			icon = GFX_situation_stage_2
			icon_background = GFX_situation_stage_frame_blue
			end = 200
			custom_tooltip = stage_eotf_dh_2_tt
			on_first_enter = {
				if = {
					limit = {
						owner = {
							is_ai = no
						}
					}
					situation_event = {
						id = eotf.260
					}
				}
			}
		}
		stage_eotf_dh_3 = {
			icon = GFX_situation_stage_3
			icon_background = GFX_situation_stage_frame_blue
			end = 300
			custom_tooltip = stage_eotf_dh_3_tt
			on_first_enter = {
				if = {
					limit = {
						owner = {
							is_ai = no
						}
					}
					situation_event = {
						id = eotf.261
					}
				}
				if = {
					limit = {
						owner = {
							is_ai = no
							NOT = { has_country_flag = eotf_dh_extremist_attack_scheduled }
						}
					}
					owner = {
						set_country_flag = eotf_dh_extremist_attack_scheduled
						country_event = {
							id = eotf.266
							days = 1
							random = 3600
						}
					}
				}
			}
		}
	}

	approach = {
		name = approach_eotf_dh_measured_inquiry
		icon = GFX_situation_approach_balance_sheet
		icon_background = GFX_situation_approach_bg_yellow
		default = yes

		potential = {
			always = yes
		}

		allow = {
			always = yes
		}

		ai_weight = {
			base = 100
		}

		on_select = {
			custom_tooltip = approach_eotf_dh_measured_inquiry_select_tt
		}

		modifier = {
			planet_stability_mult = -0.05
			country_unity_produces_mult = -0.05
			all_technology_research_speed = 0.10
		}
	}

	approach = {
		name = approach_eotf_dh_forceful_inquiry
		icon = GFX_situation_approach_research
		icon_background = GFX_situation_approach_bg_red

		potential = {
			always = yes
		}

		allow = {
			always = yes
		}

		ai_weight = {
			base = 40
		}

		on_select = {
			custom_tooltip = approach_eotf_dh_forceful_inquiry_select_tt
		}

		modifier = {
			planet_stability_mult = -0.10

		}
	}
