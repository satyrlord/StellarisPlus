building_plentiful_traditions_university_of_wisdom = {
	base_buildtime = @b2_time
	planet_limit = 1
	category = research

	building_sets = {
		research
	}

	potential = {
		owner = {
			has_tradition = tr_plentiful_experimentalism_4
		}
		NOR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_modifier = penal_colony
		}
		exists = owner
	}

	destroy_trigger = {
		exists = owner
		OR = {
			has_modifier = resort_colony
			has_modifier = slave_colony
			has_modifier = penal_colony
			owner = { is_regular_empire = no }
		}
	}

	allow = {
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b2_minerals
			exotic_gases = @b2_rare_cost
		}
		upkeep = {
			energy = @b2_upkeep
			exotic_gases = @b2_rare_upkeep
		}
		produces = {
			physics_research = 7
			society_research = 1
			engineering_research = 1
		}
	}
	
triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_regular_empire = yes
			NOT = {
				has_country_flag = fallen_empire_1
			}
			has_global_flag = plentiful_perks_active
		}
	}
	modifier = {
		job_physicist_add = 300
		job_biologist_add = 300
		job_engineer_add = 300
	}
}

triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = {
			is_hive_empire = yes
		}
		has_global_flag = plentiful_perks_active
	}
	modifier = {
		job_calculator_physicist_add = 300
		job_calculator_biologist_add = 300
		job_calculator_engineer_add = 300
	}
}

triggered_planet_modifier = {
	potential = {
		exists = owner
		owner = { is_machine_empire = yes }
		has_global_flag = plentiful_perks_active
	}
	modifier = {
		job_calculator_physicist_add = 300
		job_calculator_biologist_add = 300
		job_calculator_engineer_add = 300
	}
}
