@building_introspective_lab_cost_0 = 800
@building_introspective_lab_cost_1 = 1000
@building_introspective_lab_cost_2 = 1200
@building_introspective_lab_cost_exotic_gases_1 = 100
@building_introspective_lab_cost_exotic_gases_2 = 200
@building_introspective_lab_maintenance_0 = 4
@building_introspective_lab_maintenance_1 = 10
@building_introspective_lab_maintenance_2 = 18
@building_introspective_lab_maintenance_exotic_gases_1 = 2
@building_introspective_lab_maintenance_exotic_gases_2 = 4

building_introspective_lab = {
	base_buildtime = @b3_time
	can_demolish = yes
	
	planet_limit = 1

	exempt_from_ai_planet_specialization = no

	category = research
	
	building_sets = {
		origin
	}

	potential = {
		AND = {
			exists = owner
			owner = { has_origin = origin_orphan_matrix }
			OR = {
				has_deposit = d_introspective_lab
				has_planet_flag = d_introspective_lab
			}
		}
	}
	
	convert_to = {
		building_research_lab_1
	}

	planet_modifier = {
		job_introspective_calculator_add = 100
		pop_introspective_calculator_bonus_workforce_mult = 0.1
		planet_introspective_calculators_upkeep_mult = 0.05
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @building_introspective_lab_cost_0
		}
		upkeep = {
			energy = @building_introspective_lab_maintenance_0
		}
		upkeep = {
			trigger = {
				owner = {
					is_galactic_community_member = yes
				}
				is_active_resolution = "resolution_galacticstudies_extradimensional_experimentation"
				planet = {
					has_modifier = pm_extradimensional_experimentation
				}
			}
			sr_zro = 1
		}
	}

	upgrades = {
		building_introspective_lab_2
	}
}
