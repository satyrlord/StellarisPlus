@ascension_agenda_cost = 3500		# Same as short_agenda_cost from traditions
@ascension_agenda_cooldown = 1080	# 3 Years

agenda_sensor_4_breakthrough = {
	agenda_cost = @base_agenda_cost
	agenda_cooldown = @ascension_agenda_cooldown
	potential = {
		has_origin = origin_orphan_matrix
		has_country_flag = orphan_matrix_situation_finished
		NOT = { has_technology = tech_sensors_4 }
	}
	effect = {
		inverted_switch = {
			trigger = has_technology
			tech_sensors_4 = {
				add_tech_progress = {
					tech = tech_sensors_4
					progress = @agenda_award_tech_progress
				}
			}
			default = {
				add_monthly_resource_mult = {
					resource = physics_research
					value = @tier3researchreward
					min = @tier3researchmin
					max = @tier3researchmax
				}
			}
		}
	}
}
