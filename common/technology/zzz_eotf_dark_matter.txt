# Echoes of the Fallen: same pattern as mod "Young Fallen Empire" (workshop) — extend the
# vanilla factor=0 NOR list with has_origin, then add factor=100 for origin/flags so cards actually roll.

tech_dark_matter_deflector = {
	cost = @fallentechcost
	area = physics
	tier = @fallentechtier
	category = { field_manipulation }
	ai_update_type = all
	weight = 1
	is_rare = yes
	is_dangerous = yes

	prerequisites = { "tech_shields_5" }

	weight_modifier = {
		modifier = {
			factor = 0
			NOR = {
				is_country_type = fallen_empire
				is_country_type = awakened_fallen_empire
				has_origin = origin_ancient_legacy
				has_country_flag = eotf_command_nexus_found
				has_country_flag = eotf_design_archives_recovered
			}
		}
		modifier = {
			factor = 100
			OR = {
				has_origin = origin_ancient_legacy
				has_country_flag = eotf_command_nexus_found
				has_country_flag = eotf_design_archives_recovered
			}
		}
	}

	technology_swap = {
		name = tech_bio_dark_matter_deflector
		inherit_icon = no
		inherit_effects = no
		trigger = {
			country_uses_bio_ships = yes
		}
		weight = {
			factor = 1
		}
	}
}
