namespace = plentiful_traditions_malice

pop_event = {
	id = plentiful_traditions_malice.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		exists = planet
		NOT = {
			has_modifier = modifier_plentiful_traditions_slave1
		}
	}
	
	immediate = {
		if = {
			limit = {
				planet = { owner = { has_tradition = tr_plentiful_malice_adopt } }
			}
			add_modifier = {
				modifier = "modifier_plentiful_traditions_slave1"
				days = -1
			}
		}
	}
}

pop_event = {
	id = plentiful_traditions_malice.3
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		exists = planet
		has_modifier = pop_recently_conquered
	}
	
	immediate = {
		if = {
			limit = {
				planet = { owner = { has_tradition = tr_plentiful_malice_5 } }
			}
			remove_modifier = "pop_recently_conquered"
		}
	}
}

country_event = {
	id = plentiful_traditions_malice.4
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_tradition = tr_plentiful_malice_adopt
	}

	immediate = {
		every_owned_planet = {
			every_owned_pop_group = {
				limit = {
					is_enslaved = yes
					NOT = {
						has_modifier = modifier_plentiful_traditions_slave1
					}
				}
				add_modifier = {
					modifier = "modifier_plentiful_traditions_slave1"
					days = -1
				}
			}
		}
	}
}
