country_event = {
	id = soe.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			AND = {
				has_modifier = soe_parental_supervision
				from = { is_country_type = default }
			}
			AND = {
				from = { has_modifier = soe_parental_supervision }
				is_country_type = default
			}
		}
	}

	immediate = {
		if = {
			limit = {
				has_modifier = soe_parental_supervision
				from = { is_country_type = default }
			}
			soe_apply_parental_supervision_opinion_to_country = { TARGET = from }
		}
		if = {
			limit = {
				from = { has_modifier = soe_parental_supervision }
				is_country_type = default
			}
			from = {
				soe_apply_parental_supervision_opinion_to_country = { TARGET = root }
			}
		}
	}
}
