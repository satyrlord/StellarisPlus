# on_country_type_changed — Root = country that changed type
country_event = {
	id = soe.100
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_country_type = awakened_fallen_empire
		OR = {
			is_fanatic_materialist = yes
			is_fanatic_spiritualist = yes
		}
		OR = {
			has_country_flag = scion_master
			any_playable_country = {
				has_origin = origin_scion
				is_ai = no
				has_country_flag = soe_former_overlord@root
			}
		}
	}

	immediate = {
		every_playable_country = {
			limit = {
				has_origin = origin_scion
				is_ai = no
				NOT = { has_country_flag = soe_overlord_awakening_transmission_fired }
				OR = {
					AND = {
						exists = overlord
						overlord = {
							is_same_value = root
							soe_is_awakened_scion_master = yes
							soe_eligible_for_overlord_awakening_transmission = yes
						}
					}
					AND = {
						has_country_flag = soe_former_overlord@root
						root = { soe_eligible_for_overlord_awakening_transmission = yes }
					}
				}
			}
			soe_fire_overlord_awakening_transmission = yes
		}
	}
}

# on_war_ended — Root = Loser, From = Main Winner (no FromFrom on this hook)
country_event = {
	id = soe.121
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		soe_is_wih_active = yes
		is_country_type = awakened_fallen_empire
		OR = {
			AND = {
				NOT = { has_country_flag = scion_master }
				from = {
					is_country_type = awakened_fallen_empire
					has_country_flag = scion_master
				}
			}
			AND = {
				has_country_flag = scion_master
				from = {
					is_country_type = awakened_fallen_empire
					NOT = { has_country_flag = scion_master }
				}
			}
		}
	}

	immediate = {
		if = {
			limit = { has_country_flag = scion_master }
			save_event_target_as = soe_wih_destroyed_overlord
			soe_notify_scion_wih_fell = yes
		}
		else = {
			soe_notify_scion_wih_won = yes
		}
	}
}
