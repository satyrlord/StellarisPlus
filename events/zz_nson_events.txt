namespace = nson

# unmarks ships as boss (hidden)
# triggers via on_yearly_pulse, only performs SET every 10 years (per fleet)
event = {
	id = nson.1
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_galaxy_fleet = {
			limit = {
				NOT = { has_fleet_flag = nson_identified }
			}
			
			set_timed_fleet_flag = {
				flag = nson_identified
				years = 10
			}
			
			set_fleet_settings = {
				is_boss = no
				can_change_composition = yes
			}
		}
		
	}
}
