# Manage deficits (Because PDX didn't do it properly...)
event = {
	id = orphan_matrix_events.500
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_playable_country = {
			limit = {
				has_origin = origin_orphan_matrix
				any_owned_planet = {
					num_assigned_jobs = {
						job = introspective_calculator
						value > 0
					}
				}
			}
			
			# Stage 1
			if = {
				limit = {
					any_situation = {
						is_situation_type = situation_energy_deficit
						current_stage = deficit_first
					}
					NOT = { has_modifier = introspective_calculator_energy_deficit_1 }
				}
				
				add_modifier = {
					modifier = introspective_calculator_energy_deficit_1
					days = -1
				}
				remove_modifier = introspective_calculator_energy_deficit_2
				remove_modifier = introspective_calculator_energy_deficit_3
				remove_modifier = introspective_calculator_energy_deficit_4
			}
			
			# Stage 2
			if = {
				limit = {
					any_situation = {
						is_situation_type = situation_energy_deficit
						current_stage = deficit_second
					}
					NOT = { has_modifier = introspective_calculator_energy_deficit_2 }
				}
				
				remove_modifier = introspective_calculator_energy_deficit_1
				add_modifier = {
					modifier = introspective_calculator_energy_deficit_2
					days = -1
				}
				remove_modifier = introspective_calculator_energy_deficit_3
				remove_modifier = introspective_calculator_energy_deficit_4
			}
			
			# Stage 3
			if = {
				limit = {
					any_situation = {
						is_situation_type = situation_energy_deficit
						current_stage = deficit_third
					}
					NOT = { has_modifier = introspective_calculator_energy_deficit_3 }
				}
				
				remove_modifier = introspective_calculator_energy_deficit_1
				remove_modifier = introspective_calculator_energy_deficit_2
				add_modifier = {
					modifier = introspective_calculator_energy_deficit_3
					days = -1
				}
				remove_modifier = introspective_calculator_energy_deficit_4
			}
			
			# Stage 4
			if = {
				limit = {
					any_situation = {
						is_situation_type = situation_energy_deficit
						current_stage = deficit_fourth
					}
					NOT = { has_modifier = introspective_calculator_energy_deficit_4 }
				}
				
				remove_modifier = introspective_calculator_energy_deficit_1
				remove_modifier = introspective_calculator_energy_deficit_2
				remove_modifier = introspective_calculator_energy_deficit_3
				add_modifier = {
					modifier = introspective_calculator_energy_deficit_4
					days = -1
				}
			}
			
			# No Deficit
			if = {
				limit = {
					NOT = {
						any_situation = {
							is_situation_type = situation_energy_deficit
						}
					}
					OR = {
						has_modifier = introspective_calculator_energy_deficit_1
						has_modifier = introspective_calculator_energy_deficit_2
						has_modifier = introspective_calculator_energy_deficit_3
						has_modifier = introspective_calculator_energy_deficit_4
					}
				}
				
				if = { limit = { has_modifier = introspective_calculator_energy_deficit_1 } remove_modifier = introspective_calculator_energy_deficit_1 }
				if = { limit = { has_modifier = introspective_calculator_energy_deficit_2 } remove_modifier = introspective_calculator_energy_deficit_2 }
				if = { limit = { has_modifier = introspective_calculator_energy_deficit_3 } remove_modifier = introspective_calculator_energy_deficit_3 }
				if = { limit = { has_modifier = introspective_calculator_energy_deficit_4 } remove_modifier = introspective_calculator_energy_deficit_4 }
			}
		}
	}
}

# Upgrade Introspective Lab - I
situation_event = {
	id = orphan_matrix_events.95
	title = orphan_matrix_events.95.name
	desc = orphan_matrix_events.95.desc
	picture = GFX_evt_rare_tech_cache
	show_sound = event_administrative_work
	situation = this

	is_triggered_only = yes
	
	trigger = {
		owner = {
			NOT = { has_country_flag = lab_1_upgrade_offered }
			has_technology = tech_basic_science_lab_2
			resource_stockpile_compare = {
				resource = exotic_gases
				value > 60
			}
		}
	}
	
	immediate = {
		owner = {
			set_country_flag = alpha_codex_paused
			set_country_flag = lab_1_upgrade_offered
			capital_scope = {
				set_variable = {
					which = num_introspective_labs
					value = 0
				}
				
				export_trigger_value_to_variable = {
					trigger = num_buildings
					parameters = {
						type = building_introspective_lab
					}
					variable = num_introspective_labs
				}
			}
			
			set_variable = {
				which = introspective_labs_upgrade_1_gases
				value = 5
			}
			
			multiply_variable = {
				which = introspective_labs_upgrade_1_gases
				value = capital_scope.num_introspective_labs
			}
			
			set_variable = {
				which = introspective_labs_upgrade_1_minerals
				value = 200
			}
			
			multiply_variable = {
				which = introspective_labs_upgrade_1_minerals
				value = capital_scope.num_introspective_labs
			}
		}
	}
	
	option = {
		name = orphan_matrix_events.95.yes
		custom_tooltip = orphan_matrix_events.95.accept
		
		allow = {
			owner = {
				resource_stockpile_compare = {
					resource = minerals
					value > introspective_labs_upgrade_1_minerals
				}
				resource_stockpile_compare = {
					resource = exotic_gases
					value > introspective_labs_upgrade_1_gases
				}
			}
		}
		
		owner = {
			add_resource = {
				exotic_gases = 5
				mult = capital_scope.num_introspective_labs
			}
			add_resource = {
				minerals = 200
				mult = capital_scope.num_introspective_labs
			}
		}
		
		hidden_effect = {
			owner = { set_country_flag = lab_1_upgrade_accepted }
			
			if = {
				limit = {
					NOT = {
						owner = {
							is_variable_set = orphaned_matrix_introspective_lab_level
						}
					}
				}
				owner = {
					set_variable = {
						which = orphaned_matrix_introspective_lab_level
						value = 1
					}
				}
			}
			
			owner = {
				change_variable = {
					which = orphaned_matrix_introspective_lab_level
					value = 1
				}
			}
		}
	}
	
	option = {
		name = orphan_matrix_events.95.no
		custom_tooltip = orphan_matrix_events.95.refuse
	}
	
	after = {
		hidden_effect = {
			owner = { remove_country_flag = alpha_codex_paused }
		}
	}
}

			add_resource = { energy = -5000 }
			add_resource = { alloys = -3000 }

				modifier = motivated_drones

		name = orphan_matrix_events.97.nothing

			}

	}

		name = orphan_matrix_events.97.concede
		owner = {
			add_resource = { energy = 15000 }
			add_resource = { alloys = 8000 }
		}

			destroy_situation = this

				}
			}
		}

		ai_chance = {
			factor = 0
		}
	}
	
	after = {
		hidden_effect = {
			owner = { remove_country_flag = alpha_codex_paused }
		}
	}
